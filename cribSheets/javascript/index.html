filter()

const filteredItems = items.filter((item) => {

})

map()

const itemNames = items.map((item) => { return item.name })

find()

const foundItem = items.find((item) => { return item.name === 'Book' })

forEach()

items.forEach((item) => { console.log(item.name) })

some()

const hasInexpensiveItems = items.some((item) => { return item.price <= 100 })

every()

const hasEveryItem = items.every((item) => { return item.price <= 100 })

reduce()

const totalPrice = items.reduce((currentTotal, item) => { return item.price + currentTotal }, 0)
